#include <iostream>
using namespace std;

int main() {
    int num, hundreds, tens, ones, newNum;
    cout << "Enter a three-digit number: ";
    cin >> num;
    hundreds = num / 100; // определяем сотни
    tens = (num / 10) % 10; // определяем десятки
    ones = num % 10; // определяем единицы
    newNum = hundreds * 100 + ones * 10 + tens; // формируем новое число
    cout << "New number: " << newNum << endl;
    return 0;
}

const int N = 10; // граница ряда



int main() {

    double x;

    cout << "Введите значение x (|x| < 1): ";

    cin >> x;

    double sum = 0;

    for (int n = 0; n <= N; n++) {

    }

    cout << "Сумма ряда: " << sum << endl;

// 4 урок сам раб 4 баз 8.4 а

#include <iostream>

using namespace std;



int main() {

    int n = 5; // количество строк в таблице
    
    int arr[n][n]; // двумерный массив для хранения чисел

    
    // заполнение массива

    for (int i = 0; i < n; i++) {

        for (int j = 0; j <= i; j++) {

            if (j == 0) {

                arr[i][j] = i;

            } else {

                arr[i][j] = arr[i-1][j-1];

            }

            cout << arr[i][j] << " ";

        }

        cout << endl;

    }


    return 0;

}

#include <iostream>

using namespace std;

int main() {

    int num = 20; // начальное число

    for (int i = 5; i >= 1; i--) { // внешний цикл от 5 до 1

        for (int j = num; j <= num+i-1; j++) { // внутренний цикл от num до num+i-1

            cout << j << " "; // выводим число j и пробел

        }

        num--; // уменьшаем начальное число на 1

        cout << endl; // переходим на новую строку

    }

    return 0;

}

int main()
{
    int rows, cols;
    cout << "Enter the number of rows and columns: ";
    cin >> rows >> cols;

    // Create a matrix
    int matrix[rows][cols];

    // Input elements into the matrix
     for(int i = 0; i < rows; ++i)
        for(int j = 0; j < cols; ++j)
        {
            cout << "Enter element (" << i + 1<< "," << j + 1<< ") : ";
            cin >> matrix[i][j];
        }

   // Display original Matrix 
   cout<<"\nOriginal Matrix:\n";
      for (int i=0;i<rows;++i){
         for (int j=0;j<cols;++j){
             cout<<matrix[i][j]<<" ";  
         }
       cout<<"\n";
      }

   // Transpose of a Matrix 
   int transpose[cols][rows];
   
      for (int i=0;i<cols;++i){    
        for (int j=0;j<rows;++j){    
          transpose[i][j]=matrix[j][i];    
        }        
     }    

   // Display Transposed Matrix     
   cout<<"\nThe transposed Matrix is:\n";      
    for(int k = 0;k < cols;k++){       
       for(int l = 0;l < rows;l++){           
          cout<<transpose[k] [l]<<" ";       
        }          
      cout<<"\n";     
    } 

return(1);
}

// 5 урок сам раб 5 прод 3

#include <iostream>

#include <cmath>

using namespace std;

const int MAX_ITERATIONS = 10; // граница ряда

double cos(double x) {

    double result = 1; // значение ряда при n = 0

    double term = 1; // текущий член ряда

    for (int n = 1; n <= MAX_ITERATIONS; n++) {

        term *= -1 * x * x / ((2 * n - 1) * (2 * n));

        result += term;

    }

    return result;

}



int main() {

    double x;

    cout << "Enter x value: ";

    cin >> x;

    cout << "cos(" << x << ") = " << cos(x) << endl;

    return 0;

}
