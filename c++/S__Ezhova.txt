// Самостоятельная базовый 3, 6.13

#include <iostream>
using namespace std;

int main() {
    double a;
    cout << "Введите значение a между 1 и 1.5: ";
    cin >> a;

    int n = 2; // начинаем с n = 2
    bool all_greater_or_equal = true; // флаг, показывающий, все ли числа >= a

    while (all_greater_or_equal) {
        double sum = 1; // сумма начинается с 1
        for (int i = 2; i <= n; i++) {
            sum += 1.0 / i; // добавляем очередное слагаемое
        }
        if (sum < a) { // если хотя бы одно число меньше a
            all_greater_or_equal = false; // устанавливаем флаг в false
        } else { // иначе
            cout << "n = " << n << endl; // выводим текущее значение n
            n++; // увеличиваем n на 1
        }
    }

    return 0;
}


// Самостоятельная базовый 3, 6.14

#include <iostream>
using namespace std;

int main() {
   double a;
   cout << "Введите число a (1 < a <= 1.5): ";
   cin >> a;
   
   int n = 2; // начинаем считать сумму с 2-го числа
   double sum = 0.0; // переменная для хранения суммы
   
   while (sum < a) { // пока сумма меньше а
      sum = 1.0; // начинаем суммировать с 1
      for (int i = 2; i <= n; i++) { // считаем сумму до n-го числа
         sum += 1.0 / i; // добавляем очередное число в сумму
      }
      n++; // переходим к следующему значению n
   }
   
   cout << "Наименьшее n: " << n-1 << endl; // выводим значение n-1, так как последнее значение уже больше a
   
   return 0;
}

int main() {
   double a;

   cin >> a;
   
   int n = 2; // начинаем считать сумму с 2-го числа
   double sum = 0.0; // переменная для хранения суммы
   
   while (sum < a) { // пока сумма меньше а
      sum = 1.0; // начинаем суммировать с 1
      cout << "1 "; // выводим первое число
      for (int i = 2; i <= n; i++) { // считаем сумму до n-го числа
         sum += 1.0 / i; // добавляем очередное число в сумму
         if (sum < a) { // если сумма меньше а, выводим очередное число
            cout << "1/" << i << " ";
         }
      }
      n++; // переходим к следующему значению n
   }

// Самостоятельная базовый 3, While30

#include <iostream>
using namespace std;

int main() {
   int a, b, c;
   cout << "Введите числа A, B, C: ";
   cin >> a >> b >> c;
   
   int count = 0; // количество квадратов
   while (a >= c && b >= c) { // пока размеры прямоугольника больше стороны квадрата
      a -= c; // уменьшаем размеры прямоугольника на сторону квадрата
      b -= c;
      count++; // увеличиваем количество квадратов
   }
   
   cout << "Количество квадратов: " << count << endl;
   
   return 0;
}


// Самостоятельная продвинутый 3, задача 19

#include <iostream>
#include <cstdlib>
#include <ctime>
#include <cmath>
using namespace std;

int main() {
   double a;
   cout << "Введите параметр a: ";
   cin >> a;
   
   srand(time(NULL)); // инициализация генератора случайных чисел
   
   const int N = 1000000; // количество точек
   int count = 0; // количество точек, попавших внутрь тела
   
   for (int i = 0; i < N; i++) {
      double x = (double)rand() / RAND_MAX * a; // случайное значение x
      double y = (double)rand() / RAND_MAX * a; // случайное значение y
      if (x * y <= a * a && x + y <= 2.5 * a) { // проверка, попадает ли точка внутрь тела
         count++;
      }
   }
   
   double S = (double)count / N * a * a; // вычисление площади методом Монте-Карло
   double S_exact = (15.0/8 - 2*log(2)) * a * a; // точное значение площади
   
   cout << "Площадь, вычисленная методом Монте-Карло: " << S << endl;
   cout << "Точное значение площади: " << S_exact << endl;
   
   return 0;
}

    const int n = 10; // размер массива
    int arr[n][n]; // объявление массива

    // заполнение массива случайными числами от 1 до 10
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            arr[i][j] = rand() % 10 + 1;
        }
    }

    // вычисление суммы элементов каждой строки
    for (int i = 0; i < n; i++) {
        int sum = 0;
        for (int j = 0; j < n; j++) {
            sum += arr[i][j];
        }
        std::cout << "Сумма элементов " << i+1 << "-й строки: " << sum << std::endl;
    }

    return 0;
}


// Самостоятельная базовый 4, раздел 3 №18

#include <iostream>
#include <climits>

int main() {
    const int n = 4; // Размер массива
    int arr[n][n]; // Объявление массива
    int min_sum = INT_MAX; // Начальное значение минимальной суммы
    int min_row = -1; // Начальное значение номера строки с минимальной суммой

    // Заполнение массива и поиск строки с минимальной суммой
    for (int i = 0; i < n; i++) {
        int row_sum = 0; // Сумма элементов текущей строки
        for (int j = 0; j < n; j++) {
            std::cout << "Введите элемент [" << i << "][" << j << "]: ";
            std::cin >> arr[i][j];
            row_sum += arr[i][j]; // Добавляем элемент к сумме текущей строки
        }
        if (row_sum < min_sum) {
            min_sum = row_sum; // Обновляем значение минимальной суммы
            min_row = i; // Обновляем номер строки с минимальной суммой
        }
    }

    // Вывод результата
    std::cout << "Строка с наименьшей суммой элементов: ";
    for (int j = 0; j < n; j++) {
        std::cout << arr[min_row][j] << " ";
    }
    std::cout << std::endl;

    return 0;
}


// Самостоятельная продвинутый 4, задача 19

#include <iostream>
#include <vector>
#include <cmath>

using namespace std;

int main() {
    vector<double> x, y;
    double n, sum_x = 0, sum_y = 0, sum_xy = 0, sum_x2 = 0, a, b;

    cout << "Введите количество значений: ";
    cin >> n;

    for (int i = 0; i < n; i++) {
        double xi, yi;
        cout << "Введите значение x" << i+1 << ": ";
        cin >> xi;
        cout << "Введите значение y" << i+1 << ": ";
        cin >> yi;
        x.push_back(xi);
        y.push_back(yi);
        sum_x += xi;
        sum_y += yi;
        sum_xy += xi * yi;
        sum_x2 += pow(xi, 2);
    }

    b = (n * sum_xy - sum_x * sum_y) / (n * sum_x2 - pow(sum_x, 2));
    a = (sum_y - b * sum_x) / n;

    cout << "Параметры регрессионной модели: a = " << a << ", b = " << b << endl;

    return 0;
}
